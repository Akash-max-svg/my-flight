
// Send password reset email
export const sendPasswordResetEmail = async (email, username, resetUrl, resetToken) => {
  try {
    const transporter = createTransporter();
    if (!transporter) {
      console.log('📧 Email not configured - skipping password reset email');
      return false;
    }
    
    const content = `
      <h2 style="color: #6d28d9; margin-top: 0;">🔐 Password Reset Request</h2>
      <p style="font-size: 16px; color: #333;">Dear ${username},</p>
      <p style="font-size: 14px; color: #666; line-height: 1.6;">
        We received a request to reset your password for your Flight Booking account. 
        If you didn't make this request, you can safely ignore this email.
      </p>
      
      <p style="font-size: 14px; color: #666; line-height: 1.6; margin-bottom: 30px;">
        To reset your password, click the button below:
      </p>
      
      <div style="text-align: center; margin: 30px 0;">
        <a href="${resetUrl}" 
           style="display: inline-block; padding: 15px 40px; background: linear-gradient(135deg, #6d28d9, #4f46e5); color: white; text-decoration: none; border-radius: 12px; font-size: 16px; font-weight: bold;">
          Reset Password
        </a>
      </div>
      
      <div style="background: #f8f9fa; padding: 20px; border-radius: 8px; margin: 30px 0;">
        <p style="font-size: 14px; color: #666; margin: 0 0 10px 0;">
          If the button doesn't work, copy and paste this link into your browser:
        </p>
        <p style="font-size: 14px; color: #6d28d9; word-break: break-all; margin: 0;">
          ${resetUrl}
        </p>
      </div>
      
      <div style="background: #fff3cd; border-left: 4px solid #ffc107; padding: 15px; margin: 20px 0;">
        <p style="font-size: 14px; color: #856404; margin: 0 0 10px 0;">
          <strong>Reset Token:</strong>
        </p>
        <p style="font-size: 14px; color: #856404; font-family: monospace; margin: 0;">
          ${resetToken}
        </p>
      </div>
      
      <div style="background: #f8d7da; border-left: 4px solid #dc3545; padding: 15px; margin: 30px 0;">
        <p style="font-size: 14px; color: #721c24; margin: 0;">
          <strong>⚠️ Security Notice:</strong><br>
          This link will expire in 1 hour. If you didn't request this password reset, 
          please ignore this email or contact support if you have concerns.
        </p>
      </div>
      
      <div style="margin-top: 40px; padding-top: 20px; border-top: 1px solid #e5e7eb;">
        <p style="font-size: 14px; color: #999; margin: 0;">
          Best regards,<br>
          <strong style="color: #6d28d9;">Flight Booking Team</strong>
        </p>
      </div>
    `;
    
    const mailOptions = {
      from: `"Flight Booking System" <${process.env.EMAIL_USER}>`,
      to: email,
      subject: '🔐 Password Reset Request - Flight Booking',
      html: emailTemplate(content, 'Password Reset Request')
    };
    
    await transporter.sendMail(mailOptions);
    console.log('✅ Password reset email sent to:', email);
    return true;
  } catch (error) {
    console.error('❌ Email sending failed:', error.message);
    return false;
  }
};

export default {
  sendBookingConfirmation,
  sendCancellationEmail,
  sendFlightReminder,
  sendWelcomeEmail,
  sendBookingUpdateEmail,
  sendPasswordResetEmail,
  testEmailConfiguration,
  isEmailConfigured
};
